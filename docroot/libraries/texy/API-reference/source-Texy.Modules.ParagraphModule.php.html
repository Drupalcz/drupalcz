 30:     public function process(Texy\BlockParser $parser, $content, Texy\HtmlElement $el)
 31:     {
 32:         if ($parser->isIndented()) {
 33:             $parts = preg_split('#(\n(?! )|\n{2,})#', $content, -1, PREG_SPLIT_NO_EMPTY);
 34:         } else {
 35:             $parts = preg_split('#(\n{2,})#', $content, -1, PREG_SPLIT_NO_EMPTY);
 36:         }
 37: 
 38:         foreach ($parts as $s) {
 39:             $s = trim($s);
 40:             if ($s === '') {
 41:                 continue;
 42:             }
 43: 
 44:             // try to find modifier
 45:             $mod = NULL;
 46:             if ($mx = Regexp::match($s, '#'.Texy\Patterns::MODIFIER_H.'(?=\n|\z)#sUm', Regexp::OFFSET_CAPTURE)) {
 47:                 list($mMod) = $mx[1];
 48:                 $s = trim(substr_replace($s, '', $mx[0][1], strlen($mx[0][0])));
 49:                 if ($s === '') {
 50:                     continue;
 51:                 }
 52:                 $mod = new Texy\Modifier;
 53:                 $mod->setProperties($mMod);
 54:             }
 55: 
 56:             $res = $this->texy->invokeAroundHandlers('paragraph', $parser, [$s, $mod]);
 57:             if ($res) {
 58:                 $el->insert(NULL, $res);
 59:             }
 60:         }
 61:     }

 64:     /**
 65:      * Finish invocation.
 66:      * @return Texy\HtmlElement|FALSE
 67:      */
 68:     public function solve(Texy\HandlerInvocation $invocation, $content, Texy\Modifier $mod = NULL)
 69:     {
 70:         $texy = $this->texy;
 71: 
 72:         // find hard linebreaks
 73:         if ($texy->mergeLines) {
 74:             // ....
 75:             // ... => \r means break line
 76:             $content = Regexp::replace($content, '#\n +(?=\S)#', "\r");
 77:         } else {
 78:             $content = Regexp::replace($content, '#\n#', "\r");
 79:         }
 80: 
 81:         $el = new Texy\HtmlElement('p');
 82:         $el->parseLine($texy, $content);
 83:         $content = $el->getText(); // string
 84: 
 85:         // check content type
 86:         // block contains block tag
 87:         if (strpos($content, $texy::CONTENT_BLOCK) !== FALSE) {
 88:             $el->setName(NULL); // ignores modifier!
 89: 
 90:         // block contains text (protected)
 91:         } elseif (strpos($content, $texy::CONTENT_TEXTUAL) !== FALSE) {
 92:             // leave element p
 93: 
 94:         // block contains text
 95:         } elseif (preg_match('#[^\s'.Texy\Patterns::MARK.']#u', $content)) {
 96:             // leave element p
 97: 
 98:         // block contains only replaced element
 99:         } elseif (strpos($content, $texy::CONTENT_REPLACED) !== FALSE) {
100:             $el->setName($texy->nontextParagraph);
101: 
102:         // block contains only markup tags or spaces or nothing
103:         } else {
104:             // if {ignoreEmptyStuff} return FALSE;
105:             if (!$mod) {
106:                 $el->setName(NULL);
107:             }
108:         }
109: 
110:         if ($el->getName()) {
111:             // apply modifier
112:             if ($mod) {
113:                 $mod->decorate($texy, $el);
114:             }
115: 
116:             // add <br />
117:             if (strpos($content, "\r") !== FALSE) {
118:                 $key = $texy->protect('<br />', $texy::CONTENT_REPLACED);
119:                 $content = str_replace("\r", $key, $content);
120:             };
121:         }
122: 
123:         $content = strtr($content, "\r\n", '  ');
124:         $el->setText($content);
125: 
126:         return $el;
127:     }
