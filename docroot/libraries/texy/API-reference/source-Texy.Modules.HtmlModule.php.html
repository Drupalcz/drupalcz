 56:     /**
 57:      * Callback for: <tag attr="...">.
 58:      * @return HtmlElement|string|FALSE
 59:      */
 60:     public function patternTag(Texy\LineParser $parser, array $matches)
 61:     {
 62:         list(, $mEnd, $mTag, $mAttr, $mEmpty) = $matches;
 63:         // [1] => /
 64:         // [2] => tag
 65:         // [3] => attributes
 66:         // [4] => /
 67: 
 68:         $isStart = $mEnd !== '/';
 69:         $isEmpty = $mEmpty === '/';
 70:         if (!$isEmpty && substr($mAttr, -1) === '/') { // uvizlo v $mAttr?
 71:             $mAttr = substr($mAttr, 0, -1);
 72:             $isEmpty = TRUE;
 73:         }
 74: 
 75:         // error - can't close empty element
 76:         if ($isEmpty && !$isStart) {
 77:             return FALSE;
 78:         }
 79: 
 80:         // error - end element with atttrs
 81:         $mAttr = trim(strtr($mAttr, "\n", ' '));
 82:         if ($mAttr && !$isStart) {
 83:             return FALSE;
 84:         }
 85: 
 86:         $el = new HtmlElement($mTag);
 87: 
 88:         if ($isStart) {
 89:             // parse attributes
 90:             $matches2 = NULL;
 91:             preg_match_all(
 92:                 '#([a-z0-9:-]+)\s*(?:=\s*(\'[^\']*\'|"[^"]*"|[^\'"\s]+))?()#isu',
 93:                 $mAttr,
 94:                 $matches2,
 95:                 PREG_SET_ORDER
 96:             );
 97: 
 98:             foreach ($matches2 as $m) {
 99:                 $key = strtolower($m[1]);
100:                 $value = $m[2];
101:                 if ($value == NULL) {
102:                     $el->attrs[$key] = TRUE;
103:                 } elseif ($value{0} === '\'' || $value{0} === '"') {
104:                     $el->attrs[$key] = html_entity_decode(substr($value, 1, -1), ENT_QUOTES, 'UTF-8');
105:                 } else {
106:                     $el->attrs[$key] = html_entity_decode($value, ENT_QUOTES, 'UTF-8');
107:                 }
108:             }
109:         }
110: 
111:         $res = $this->texy->invokeAroundHandlers('htmlTag', $parser, [$el, $isStart, $isEmpty]);
112: 
113:         if ($res instanceof HtmlElement) {
114:             return $this->texy->protect($isStart ? $res->startTag() : $res->endTag(), $res->getContentType());
115:         }
116: 
117:         return $res;
118:     }

121:     /**
122:      * Finish invocation.
123:      * @return HtmlElement|string|FALSE
124:      */
125:     public function solveTag(Texy\HandlerInvocation $invocation, HtmlElement $el, $isStart, $forceEmpty = NULL)
126:     {
127:         $texy = $this->texy;
128: 
129:         // tag & attibutes
130:         $allowedTags = $texy->allowedTags; // speed-up
131:         if (!$allowedTags) {
132:             return FALSE; // all tags are disabled
133:         }
134: 
135:         // convert case
136:         $name = $el->getName();
137:         $lower = strtolower($name);
138:         if (isset($texy->dtd[$lower]) || $name === strtoupper($name)) {
139:             // complete UPPER convert to lower
140:             $name = $lower;
141:             $el->setName($name);
142:         }
143: 
144:         if (is_array($allowedTags)) {
145:             if (!isset($allowedTags[$name])) {
146:                 return FALSE;
147:             }
148:             $allowedAttrs = $allowedTags[$name]; // allowed attrs
149: 
150:         } else {
151:             // allowedTags === Texy\Texy::ALL
152:             if ($forceEmpty) {
153:                 $el->setName($name, TRUE);
154:             }
155:             $allowedAttrs = $texy::ALL; // all attrs are allowed
156:         }
157: 
158:         // end tag? we are finished
159:         if (!$isStart) {
160:             return $el;
161:         }
162: 
163:         $elAttrs = & $el->attrs;
164: 
165:         // process attributes
166:         if (!$allowedAttrs) {
167:             $elAttrs = [];
168: 
169:         } elseif (is_array($allowedAttrs)) {
170: 
171:             // skip disabled
172:             $allowedAttrs = array_flip($allowedAttrs);
173:             foreach ($elAttrs as $key => $foo) {
174:                 if (!isset($allowedAttrs[$key])) {
175:                     unset($elAttrs[$key]);
176:                 }
177:             }
178:         }
179: 
180:         // apply allowedClasses
181:         $tmp = $texy->_classes; // speed-up
182:         if (isset($elAttrs['class'])) {
183:             if (is_array($tmp)) {
184:                 $elAttrs['class'] = explode(' ', $elAttrs['class']);
185:                 foreach ($elAttrs['class'] as $key => $value) {
186:                     if (!isset($tmp[$value])) {
187:                         unset($elAttrs['class'][$key]); // id & class are case-sensitive
188:                     }
189:                 }
190: 
191:             } elseif ($tmp !== $texy::ALL) {
192:                 $elAttrs['class'] = NULL;
193:             }
194:         }
195: 
196:         // apply allowedClasses for ID
197:         if (isset($elAttrs['id'])) {
198:             if (is_array($tmp)) {
199:                 if (!isset($tmp['#' . $elAttrs['id']])) {
200:                     $elAttrs['id'] = NULL;
201:                 }
202:             } elseif ($tmp !== $texy::ALL) {
203:                 $elAttrs['id'] = NULL;
204:             }
205:         }
206: 
207:         // apply allowedStyles
208:         if (isset($elAttrs['style'])) {
209:             $tmp = $texy->_styles; // speed-up
210:             if (is_array($tmp)) {
211:                 $styles = explode(';', $elAttrs['style']);
212:                 $elAttrs['style'] = NULL;
213:                 foreach ($styles as $value) {
214:                     $pair = explode(':', $value, 2);
215:                     $prop = trim($pair[0]);
216:                     if (isset($pair[1]) && isset($tmp[strtolower($prop)])) { // CSS is case-insensitive
217:                         $elAttrs['style'][$prop] = $pair[1];
218:                     }
219:                 }
220:             } elseif ($tmp !== $texy::ALL) {
221:                 $elAttrs['style'] = NULL;
222:             }
223:         }
224: 
225:         foreach (['src', 'href', 'name', 'id'] as $attr) {
226:             if (isset($elAttrs[$attr])) {
227:                 $elAttrs[$attr] = is_string($elAttrs[$attr]) ? trim($elAttrs[$attr]) : '';
228:                 if ($elAttrs[$attr] === '') {
229:                     unset($elAttrs[$attr]);
230:                 }
231:             }
232:         }
233: 
234:         if ($name === 'img') {
235:             if (!isset($elAttrs['src']) || !$texy->checkURL($elAttrs['src'], $texy::FILTER_IMAGE)) {
236:                 return FALSE;
237:             }
238:             $texy->summary['images'][] = $elAttrs['src'];
239: 
240:         } elseif ($name === 'a') {
241:             if (!isset($elAttrs['href']) && !isset($elAttrs['name']) && !isset($elAttrs['id'])) {
242:                 return FALSE;
243:             }
244:             if (isset($elAttrs['href'])) {
245:                 if ($texy->linkModule->forceNoFollow && strpos($elAttrs['href'], '//') !== FALSE) {
246:                     if (isset($elAttrs['rel'])) {
247:                         $elAttrs['rel'] = (array) $elAttrs['rel'];
248:                     }
249:                     $elAttrs['rel'][] = 'nofollow';
250:                 }
251: 
252:                 if (!$texy->checkURL($elAttrs['href'], $texy::FILTER_ANCHOR)) {
253:                     return FALSE;
254:                 }
255: 
256:                 $texy->summary['links'][] = $elAttrs['href'];
257:             }
258: 
259:         } elseif (preg_match('#^h[1-6]#i', $name)) {
260:             $texy->headingModule->TOC[] = [
261:                 'el' => $el,
262:                 'level' => (int) substr($name, 1),
263:                 'type' => 'html',
264:             ];
265:         }
266: 
267:         $el->validateAttrs($texy->dtd);
268: 
269:         return $el;
270:     }

277:     public function solveComment(Texy\HandlerInvocation $invocation, $content)
278:     {
279:         if (!$this->passComment) {
280:             return '';
281:         }
282:

284:         $content = Texy\Regexp::replace($content, '#-{2,}#', ' - ');
