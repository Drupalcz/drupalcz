158:         } else {

203:     public function getReference($name)

205:         $name = Texy\Utf::strtolower($name);

207:             return clone $this->references[$name];

249:             $image = $texy->imageModule->getReference($dest);

252:                 $link->modifier = $image->modifier;

277:     {

280:         }

284:         $el = new Texy\HtmlElement('a');
285: 
286:         if (empty($link->modifier)) {

288:         } else {
289:             $nofollow = isset($link->modifier->classes['nofollow']);
290:             $popup = isset($link->modifier->classes['popup']);
291:             unset($link->modifier->classes['nofollow'], $link->modifier->classes['popup']);
292:             $el->attrs['href'] = NULL; // trick - move to front
293:             $link->modifier->decorate($texy, $el);
294:         }
295: 
296:         if ($link->type === Link::IMAGE) {
297:             // image
298:             $el->attrs['href'] = Texy\Helpers::prependRoot($link->URL, $texy->imageModule->linkedRoot);
299:             if ($this->imageClass) {
300:                 $el->attrs['class'][] = $this->imageClass;
301:             } else {
302:                 $el->attrs['onclick'] = $this->imageOnClick;
303:             }
304: 
305:         } else {
306:             $el->attrs['href'] = Texy\Helpers::prependRoot($link->URL, $this->root);
307: 
308:             // rel="nofollow"
309:             if ($nofollow || ($this->forceNoFollow && strpos($el->attrs['href'], '//') !== FALSE)) {
310:                 $el->attrs['rel'] = 'nofollow';
311:             }
312:         }
313: 
314:         // popup on click
315:         if ($popup) {
316:             $el->attrs['onclick'] = $this->popupOnClick;
317:         }
318: 
319:         if ($content !== NULL) {
320:             $el->add($content);
321:         }
322: 
323:         $texy->summary['links'][] = $el->attrs['href'];
324: 
325:         return $el;
326:     }

352:     /**
353:      * Checks and corrects $URL.
354:      * @return void
355:      */
356:     private function checkLink(Link $link)
357:     {
358:         // remove soft hyphens; if not removed by Texy\Texy::process()
359:         $link->URL = str_replace("\xC2\xAD", '', $link->URL);
360: 
361:         if (strncasecmp($link->URL, 'www.', 4) === 0) {
362:             // special supported case
363:             $link->URL = 'http://' . $link->URL;
364: 
365:         } elseif (preg_match('#'.self::$EMAIL.'$#Au', $link->URL)) {
366:             // email
367:             $link->URL = 'mailto:' . $link->URL;
368: 
369:         } elseif (!$this->texy->checkURL($link->URL, Texy\Texy::FILTER_ANCHOR)) {
370:             $link->URL = NULL;
371: 
372:         } else {
373:             $link->URL = str_replace('&amp;', '&', $link->URL); // replace unwanted &amp;
374:         }
375:     }

378:     /**
379:      * Returns textual representation of URL.
380:      * @return string
381:      */
382:     private function textualUrl(Link $link)
383:     {
384:         if ($this->texy->obfuscateEmail && preg_match('#^'.self::$EMAIL.'$#u', $link->raw)) { // email
385:             return str_replace('@', '&#64;<!-- -->', $link->raw);
386:         }
387: 
388:         if ($this->shorten && preg_match('#^(https?://|ftp://|www\.|/)#i', $link->raw)) {
389: 
390:             $raw = strncasecmp($link->raw, 'www.', 4) === 0 ? 'none://' . $link->raw : $link->raw;
391: 
392:             // parse_url() in PHP damages UTF-8 - use regular expression
393:             if (!preg_match('~^(?:(?P<scheme>[a-z]+):)?(?://(?P<host>[^/?#]+))?(?P<path>(?:/|^)(?!/)[^?#]*)?(?:\?(?P<query>[^#]*))?(?:#(?P<fragment>.*))?()$~', $raw, $parts)) {
394:                 return $link->raw;
395:             }
396: 
397:             $res = '';
398:             if ($parts['scheme'] !== '' && $parts['scheme'] !== 'none') {
399:                 $res .= $parts['scheme'] . '://';
400:             }
401: 
402:             if ($parts['host'] !== '') {
403:                 $res .= $parts['host'];
404:             }
405: 
406:             if ($parts['path'] !== '') {
407:                 $res .= (iconv_strlen($parts['path'], 'UTF-8') > 16 ? ("/\xe2\x80\xa6" . iconv_substr($parts['path'], -12, 12, 'UTF-8')) : $parts['path']);
408:             }
409: 
410:             if ($parts['query'] !== '') {
411:                 $res .= iconv_strlen($parts['query'], 'UTF-8') > 4 ? "?\xe2\x80\xa6" : ('?' . $parts['query']);
412:             } elseif ($parts['fragment'] !== '') {
413:                 $res .= iconv_strlen($parts['fragment'], 'UTF-8') > 4 ? "#\xe2\x80\xa6" : ('#' . $parts['fragment']);
414:             }
415:             return $res;
416:         }
417: 
418:         return $link->raw;
419:     }
